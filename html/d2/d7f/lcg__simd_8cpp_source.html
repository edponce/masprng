114 void VLCG::multiply(SIMD_INT * const a, SIMD_INT * const b, const SIMD_INT c) const
115 {
116  SIMD_INT s[4], res[4];
117  SIMD_INT vtmp, vtmp2;
118 
119  s[0] = simd_set(4095);
120  s[0] = simd_and(a[1], s[0]);
121  s[1] = simd_srl_32(a[1], 0xc);
122  s[2] = simd_set(4095);
123  s[2] = simd_and(a[0], s[2]);
124  s[3] = simd_srl_32(a[0], 0xc);
125 
126  res[0] = simd_mullo_i32(b[0], s[0]);
127  res[1] = simd_mullo_i32(b[1], s[0]);
128  vtmp = simd_mullo_i32(b[0], s[1]);
129  res[1] = simd_add_i32(res[1], vtmp);
130  res[2] = simd_mullo_i32(b[2], s[0]);
131  vtmp = simd_mullo_i32(b[1], s[1]);
132  res[2] = simd_add_i32(res[2], vtmp);
133  vtmp = simd_mullo_i32(b[0], s[2]);
134  res[2] = simd_add_i32(res[2], vtmp);
135  res[3] = simd_mullo_i32(b[3], s[0]);
136  vtmp = simd_mullo_i32(b[2], s[1]);
137  res[3] = simd_add_i32(res[3], vtmp);
138  vtmp = simd_mullo_i32(b[1], s[2]);
139  res[3] = simd_add_i32(res[3], vtmp);
140  vtmp = simd_mullo_i32(b[0], s[3]);
141  res[3] = simd_add_i32(res[3], vtmp);
142 
143  vtmp = simd_srl_32(a[1], 0x18);
144  vtmp = simd_add_i32(vtmp, res[2]);
145  vtmp2 = simd_srl_32(res[1], 0xc);
146  vtmp = simd_add_i32(vtmp, vtmp2);
147  vtmp2 = simd_sll_32(res[3], 0xc);
148  a[0] = simd_add_i32(vtmp, vtmp2);
149 
150  vtmp = simd_set(4095);
151  vtmp = simd_and(res[1], vtmp);
152  vtmp = simd_sll_32(vtmp, 0xc);
153  vtmp = simd_add_i32(res[0], vtmp);
154  a[1] = simd_add_i32(vtmp, c);
155 
156  vtmp = simd_set(16777215);
157  a[0] = simd_and(a[0], vtmp);
158  a[1] = simd_and(a[1], vtmp);
159 }

168 int VLCG::init_rng(int gn, int tg, int * const s, int * const m)
169 {
170  if (tg <= 0) {
171  printf("ERROR: total_gen out of range, %d\n", tg);
172  tg = 1;
173  }
174 
175  if (gn < 0 || gn >= tg || gn >= CONFIG.LCG_MAX_STREAMS) {
176  printf("ERROR: gennum out of range, %d\n", gn);
177  return -1;
178  }
179  const int need = 1;
180  int lprime;
181  getprime_32(need, &lprime, gn);
182 
183  for (int i = 0; i < SIMD_STREAMS_MULT; ++i) {
184  if (m[i] < 0 || m[i] >= CONFIG.NPARAMS) {
185  printf("ERROR: multiplier out of range, %d\n", m[i]);
186  m[i] = 0;
187  }
188  }
189 
190  SIMD_INT vs;
191 #if defined(LONG_SPRNG)
192  parameter = simd_set(m, SIMD_STREAMS_MULT);
193  vs = simd_set(s, SIMD_STREAMS_SEED);
194  init_seed = simd_and(vs, vmsk_lsb31);
195 
196  prime_next = simd_set_64(tg);
197  prime_position = simd_set_64(gn);
198 
199  long int lmultiplier[SIMD_STREAMS_MULT];
200  for (int i = 0; i < SIMD_STREAMS_MULT; ++i)
201  lmultiplier[i] = CONFIG.MULT[m[i]];
202  multiplier = simd_set(lmultiplier, SIMD_STREAMS_MULT);
203 
204  vs = simd_sll_64(init_seed, 0x10);
205  seed = simd_set(CONFIG.INIT_SEED);
206  seed = simd_xor(seed, vs);
207 
208  prime = simd_set_64(lprime);
209  for (int i = 0; i < SIMD_STREAMS_SEED; ++i)
210  if (simd_test_zero(prime, vmsk_lh64[i]) == 1)
211  seed = simd_or(seed, vmsk_lsb1[i]);
212 #else
213  parameter = simd_set(m[3], m[2], m[1], m[0]);
214  vs = simd_set(s[3], s[2], s[1], s[0]);
215  init_seed = simd_and(vs, vmsk_lsb31);
216 
217  prime_next = simd_set(tg);
218  prime_position = simd_set(gn);
219 
220  for (int i = 0; i < SIMD_STREAMS_MULT; ++i)
221  multiplier[i] = simd_load(CONFIG.MULT[m[i]]);
222 
223  SIMD_INT vtmp;
224  seed[0] = simd_set(CONFIG.INIT_SEED[0]);
225  vtmp = simd_srl_32(init_seed, 0x8);
226  vtmp = simd_and(vtmp, vmsk_lsb24);
227  seed[0] = simd_or(seed[0], vtmp);
228 
229  simd_print("seed[0]", seed[0]);
230 
231  seed[1] = simd_set(CONFIG.INIT_SEED[1]);
232  vtmp = simd_sll_32(init_seed, 0x10);
233  vtmp = simd_and(vtmp, vmsk_msbset24);
234  seed[1] = simd_or(seed[1], vtmp);
235 
236  simd_print("seed[1]", seed[1]);
237 
238  prime = simd_set(lprime);
239  // NOTE: prime check/set is incomplete
240 #endif
241 
242  // Run generator for a while, same runs for each stream
243  for (int i = 0; i < (CONFIG.LCG_RUNUP * gn); ++i)
244  get_rn_dbl();
245 
246  return 0;
247 }
