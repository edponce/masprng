68 void LCG::multiply(int * const a, const int * const b, const int c) const
69 {
70  int s[4], res[4];
71 
72  s[0] = a[1] & 4095;
73  s[1] = a[1] >> 12;
74  s[2] = a[0] & 4095;
75  s[3] = (unsigned int)a[0] >> 12;
76 
77  res[0] = b[0] * s[0];
78  res[1] = b[1] * s[0] + b[0] * s[1];
79  res[2] = b[2] * s[0] + b[1] * s[1] + b[0] * s[2];
80  res[3] = b[3] * s[0] + b[2] * s[1] + b[1] * s[2] + b[0] * s[3];
81 
82  a[0] = ((unsigned int)a[1] >> 24) + res[2] + ((unsigned int)res[1] >> 12) + (res[3] << 12);
83  a[1] = res[0] + ((res[1] & 4095) << 12) + c;
84 
85  a[0] &= 16777215;
86  a[1] &= 16777215;
87 }
