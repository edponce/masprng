72 inline SIMD_INT simd_mul_u64(const SIMD_INT va, const SIMD_INT vb)
73 {
74  SIMD_INT vshf, vh;
75  vshf = _mm_shuffle_epi32(vb, 0xB1); // shuffle multiplier
76  vh = _mm_mullo_epi32(va, vshf); // xl * yh, xh * yl
77  vshf = _mm_slli_epi64(vh, 0x20); // shift << 32
78  vh = _mm_add_epi64(vh, vshf); // h = h1 + h2
79  vshf = _mm_set1_epi64x(0xFFFFFFFF00000000UL);
80  vh = _mm_and_si128(vh, vshf); // h & 0xFFFFFFFF00000000
81  const SIMD_INT vl = _mm_mul_epu32(va, vb); // l = xl * yl
82  return _mm_add_epi64(vl, vh); // l + h
83 }
